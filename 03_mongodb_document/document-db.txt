// 1. Retrieve information on reservations made by Lucas
db.hotels.find(
  {
    "rooms.bookings.guests": {
      $elemMatch: {
        guest_name: "Lucas",
        made_reservation: "yes"
      }
    }
  },
  {
    hotel_name: 1,
    "rooms.room_id": 1,
    "rooms.bookings": 1,
    _id: 0
  }
)

// 2. Enrika cancelled her reservation, this data needs to be removed from the database.
db.hotels.updateMany(
  {},
  {
    $pull: {
      "rooms.$[].bookings": {
        guests: {
          $elemMatch: {
            guest_name: "Enrika",
            made_reservation: "yes"
          }
        }
      }
    }
  }
)

// 3. Add a note to the reservation made by Lucas that he will arrive at the hotel after 10 pm.
db.hotels.updateOne(
  {
    "rooms.bookings.guests": {
      $elemMatch: {
        guest_name: "Lucas",
        made_reservation: "yes"
      }
    }
  },
  [
    {
      $set: {
        "rooms": {
          $map: {
            input: "$rooms",
            as: "room",
            in: {
              $mergeObjects: [
                "$$room",
                {
                  bookings: {
                    $map: {
                      input: "$$room.bookings",
                      as: "booking",
                      in: {
                        $cond: [
                          {
                            $in: [
                              {
                                guest_name: "Lucas",
                                made_reservation: "yes"
                              },
                              "$$booking.guests"
                            ]
                          },
                          {
                            $mergeObjects: [
                              "$$booking",
                              {
                                notes: {
                                  $concat: [
                                    { $ifNull: ["$$booking.notes", ""] },
                                    "; Arrives after 10pm"
                                  ]
                                }
                              }
                            ]
                          },
                          "$$booking"
                        ]
                      }
                    }
                  }
                }
              ]
            }
          }
        }
      }
    }
  ]
)

// 4. In the 'Redisson' hotel, all rooms have had their 'TVs' replaced with 'Flat-screen TVs'. Update the
information in the database accordingly.
db.hotels.updateOne(
  { hotel_name: "Redisson" },
  {
    $set: {
      "rooms": {
        $map: {
          input: "$rooms",
          as: "room",
          in: {
            $mergeObjects: [
              "$$room",
              {
                facilities: {
                  $map: {
                    input: "$$room.facilities",
                    as: "facility",
                    in: {
                      $cond: [
                        { $eq: ["$$facility", "TV"] },
                        "Flat-screen TV",
                        "$$facility"
                      ]
                    }
                  }
                }
              }
            ]
          }
        }
      }
    }
  }
)

// 5. Calculate how many bookings were made in each hotel.
db.hotels.aggregate([
  {
    $project: {
      hotel_name: 1,
      total_bookings: {
        $sum: {
          $map: {
            input: "$rooms",
            as: "room",
            in: { $size: "$$room.bookings" }
          }
        }
      },
      rooms: 0
    }
  }
])
